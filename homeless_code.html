fig, ax1 = plt.subplots(figsize=(10, 6))

# Plot R scores
ax1.set_title('Model Performance Comparison')
ax1.set_xlabel('Models')
ax1.set_ylabel('R Score', color='tab:blue')
ax1.bar(model_names, r2_values, color='tab:blue', alpha=0.6, label='R Score')
ax1.tick_params(axis='y', labelcolor='tab:blue')

# Create a second y-axis for MSE
ax2 = ax1.twinx()
ax2.set_ylabel('MSE', color='tab:red')
ax2.plot(model_names, mse_values, color='tab:red', marker='o', linestyle='-', label='MSE')
ax2.tick_params(axis='y', labelcolor='tab:red')

# Add legend and show the plot
fig.tight_layout()
fig.legend(loc='upper right')
